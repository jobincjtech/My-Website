💾 Backup & Restore Data

☁ Cloud Sync Setup (Google Sheets)

📚 One-time setup (5 minutes):
Step 1: Go to sheets.google.com and create a new spreadsheet
Step 2: Click Extensions → Apps Script
Step 3: Delete existing code, paste the code below, click Save
Step 4: Click Deploy → New Deployment → Web App
Step 5: Set Execute as: Me  |  Who has access: Anyone
Step 6: Click Deploy, copy the Web App URL, paste below
function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Data") || SpreadsheetApp.getActiveSpreadsheet().insertSheet("Data"); var data = {}; var rows = sheet.getDataRange().getValues(); for (var i = 0; i < rows.length; i++) { if (rows[i][0]) data[rows[i][0]] = rows[i][1]; } return ContentService.createTextOutput(JSON.stringify({data: data})).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Data") || SpreadsheetApp.getActiveSpreadsheet().insertSheet("Data"); var payload = JSON.parse(e.postData.contents); if (payload.action === "save") { sheet.clearContents(); var rows = Object.keys(payload.data).map(function(k) { return [k, payload.data[k]]; }); if (rows.length) sheet.getRange(1, 1, rows.length, 2).setValues(rows); } return ContentService.createTextOutput("OK"); }
Your data is stored in your own Google Sheet — Anthropic and GitHub cannot access it.
